    Create src/components/WindowPresetSwitcher.tsx with:

    ```tsx
    import { useState } from "react";
    import { AnimatePresence, motion } from "motion/react";
    import { WINDOW_PRESETS, type WindowPresetId } from "@/lib/windowPresets";
    import { setCurrentWindowPreset } from "@/canvas/tools/windowTool";
    import { useReducedMotion } from "@/hooks/useReducedMotion";
    import { springTransition } from "@/lib/motion";
    import { Button } from "@/components/ui/Button";
    import { Input } from "@/components/ui/Input";
    import { cn } from "@/lib/cn";

    type ActiveId = WindowPresetId | "custom";

    export function WindowPresetSwitcher() {
      const reduced = useReducedMotion();
      const [activeId, setActiveId] = useState<ActiveId>("standard");
      const [customDims, setCustomDims] = useState({ width: 3, height: 4, sillHeight: 3 });

      const handleChipClick = (id: ActiveId) => {
        setActiveId(id);
        if (id === "custom") {
          setCurrentWindowPreset(customDims);
        } else {
          const p = WINDOW_PRESETS.find((x) => x.id === id);
          if (p) {
            const dims = { width: p.width, height: p.height, sillHeight: p.sillHeight };
            setCurrentWindowPreset(dims);
            setCustomDims(dims); // D-06: Custom defaults to last-used preset
          }
        }
      };

      const handleCustomChange = (field: "width" | "height" | "sillHeight", v: number) => {
        const next = { ...customDims, [field]: v };
        setCustomDims(next);
        setCurrentWindowPreset(next); // D-04: live ghost update
      };

      return (
        <div
          data-testid="window-preset-switcher"
          className="fixed bottom-20 left-1/2 -translate-x-1/2 z-40 flex flex-col items-center gap-2"
        >
          <div className={cn(
            "flex items-center gap-1 px-2 py-1.5",
            "bg-card/95 backdrop-blur border border-border rounded-smooth-lg shadow-lg",
            "font-sans"
          )}>
            {WINDOW_PRESETS.map((p) => (
              <Button
                key={p.id}
                data-testid={`window-preset-chip-${p.id}`}
                variant="ghost"
                size="sm"
                onClick={() => handleChipClick(p.id)}
                className={cn(
                  activeId === p.id && "bg-accent/10 ring-1 ring-accent/40"
                )}
              >
                {p.label}
              </Button>
            ))}
            <Button
              data-testid="window-preset-chip-custom"
              variant="ghost"
              size="sm"
              onClick={() => handleChipClick("custom")}
              className={cn(
                activeId === "custom" && "bg-accent/10 ring-1 ring-accent/40"
              )}
            >
              Custom
            </Button>
          </div>

          <AnimatePresence>
            {activeId === "custom" && (
              <motion.div
                initial={reduced ? false : { opacity: 0, y: -4 }}
                animate={{ opacity: 1, y: 0 }}
                exit={reduced ? { opacity: 0 } : { opacity: 0, y: -4 }}
                transition={reduced ? { duration: 0 } : springTransition(reduced)}
                className={cn(
                  "flex items-center gap-2 px-3 py-2",
                  "bg-card/95 backdrop-blur border border-border rounded-smooth-md shadow-lg font-sans text-sm"
                )}
              >
                <label className="flex items-center gap-1">
                  <span className="text-muted-foreground">W</span>
                  <Input
                    data-testid="window-preset-custom-width"
                    type="number"
                    step={0.5}
                    value={customDims.width}
                    onChange={(e) => handleCustomChange("width", Number(e.target.value))}
                    className="w-16"
                  />
                </label>
                <label className="flex items-center gap-1">
                  <span className="text-muted-foreground">H</span>
                  <Input
                    data-testid="window-preset-custom-height"
                    type="number"
                    step={0.5}
                    value={customDims.height}
                    onChange={(e) => handleCustomChange("height", Number(e.target.value))}
                    className="w-16"
                  />
                </label>
                <label className="flex items-center gap-1">
                  <span className="text-muted-foreground">Sill</span>
                  <Input
                    data-testid="window-preset-custom-sill"
                    type="number"
                    step={0.5}
                    value={customDims.sillHeight}
                    onChange={(e) => handleCustomChange("sillHeight", Number(e.target.value))}
                    className="w-16"
                  />
                </label>
              </motion.div>
            )}
          </AnimatePresence>
        </div>
      );
    }
    ```

    If the project's Button/Input prop shapes differ (e.g., they don't accept `data-testid` via spread), adapt — wrap in a `<span data-testid>` or extend Props. Verify against the actual primitive source.
