    GREEN: finalize src/three/ProductMesh.tsx with the full branching logic per D-08 + 56-RESEARCH.md §5:

    ```typescript
    import { Suspense } from "react";
    import { ErrorBoundary } from "react-error-boundary";   // NOT inline class (D-05)
    import { useGltfBlobUrl } from "@/hooks/useGltfBlobUrl";
    import { ProductBox } from "./ProductBox";
    import GltfProduct from "./GltfProduct";

    export default function ProductMesh({ placed, product, isSelected }: Props) {
      const { handlePointerDown, handlePointerUp } = useClickDetect(...);
      const { width, depth, height, isPlaceholder } = resolveEffectiveDims(product, placed);
      const textureUrl = !isPlaceholder && product?.imageUrl ? product.imageUrl : null;
      const texture = useProductTexture(textureUrl);
      const rotY = -(placed.rotation * Math.PI) / 180;
      const { url } = useGltfBlobUrl(product?.gltfId);   // always called (rules of hooks)

      const handlers = { onPointerDown, onPointerUp, onContextMenu: ... };

      if (product?.gltfId) {
        return (
          <group position={[placed.position.x, 0, placed.position.y]} rotation={[0, rotY, 0]} {...handlers}>
            {url ? (
              <Suspense fallback={<ProductBox ... />}>
                <ErrorBoundary fallback={<ProductBox ... />}>
                  <GltfProduct url={url} width={width} depth={depth} height={height} isSelected={isSelected} />
                </ErrorBoundary>
              </Suspense>
            ) : (
              <ProductBox ... />   // IDB fetch in flight OR error (D-04, D-05)
            )}
          </group>
        );
      }

      // Image-only / placeholder — unchanged (D-12 regression guard)
      return (
        <group position={[placed.position.x, 0, placed.position.y]} rotation={[0, rotY, 0]} {...handlers}>
          <ProductBox width={width} depth={depth} height={height}
            isSelected={isSelected} isPlaceholder={isPlaceholder} textureUrl={textureUrl} />
        </group>
      );
    }
    ```

    CRITICAL: useGltfBlobUrl must always be called (hooks rule) even on the image-only path. The url result is only used inside `if (product?.gltfId)`.

    Suspense wraps ErrorBoundary (not reverse) — per 56-RESEARCH.md §5: Suspense catches thrown Promises (loading); ErrorBoundary catches thrown Errors (failures).

    Regression check after GREEN: `npm run test:quick` — pre-existing failures unchanged, 3 new component tests passing, all existing ProductMesh tests still passing.
