import { describe, it, expect } from "vitest";
import { buildTree, isEffectivelyHidden } from "./treeBuilder";
import type { RoomDoc, WallSegment } from "@/types/cad";

const mkWall = (id: string, sx: number, sy: number, ex: number, ey: number): WallSegment => ({
  id, start: { x: sx, y: sy }, end: { x: ex, y: ey }, thickness: 0.5, openings: [],

const mkRoom = (id: string, walls: Record<string, WallSegment> = {}): RoomDoc => ({

  walls,
  placedProducts: {},
  placedCustomElements: {},
  ceilings: {},

describe("buildTree", () => {
  it("empty rooms → empty tree", () => {
    expect(buildTree({}, {}, [])).toEqual([]);
  });
  it("one room with one wall produces room → walls-group → wall-leaf", () => {
    const tree = buildTree({ r1: mkRoom("r1", { w1: mkWall("w1", 0, 0, 10, 0) }) }, {}, []);

    expect(tree[0].kind).toBe("room");
    expect(tree[0].children?.[0].kind).toBe("group");
    expect(tree[0].children?.[0].id).toBe("r1:walls");
    expect(tree[0].children?.[0].children).toHaveLength(1);
    expect(tree[0].children?.[0].children?.[0].kind).toBe("wall");

  it("group is omitted when empty", () => {
    const tree = buildTree({ r1: mkRoom("r1") }, {}, []);
    expect(tree[0].children).toEqual([]); // or undefined per impl — adjust in Plan 02

});

describe("isEffectivelyHidden", () => {
  it("returns false on empty hiddenIds", () => {
    expect(isEffectivelyHidden(["r1", "r1:walls", "w1"], new Set())).toBe(false);
  });
  it("returns true if any ancestor is hidden", () => {
