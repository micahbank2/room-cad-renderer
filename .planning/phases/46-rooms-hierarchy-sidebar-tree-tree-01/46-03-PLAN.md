Render structure (font-mono throughout):
```
<div data-tree-node={node.id} data-tree-kind={node.kind}>
  <div className={`flex items-center gap-1 ${depth === 0 ? "" : depth === 1 ? "pl-2" : depth === 2 ? "pl-4" : "pl-6"} py-1`}>
    {hasChildren && (
      <button data-tree-chevron onClick={() => onToggleExpand(node.id)} aria-label={isOpen ? "Collapse" : "Expand"} aria-expanded={isOpen} className="text-text-ghost hover:text-accent">
        {isOpen ? <ChevronDown size={12}/> : <ChevronRight size={12}/>}
      </button>
    )}
    <button data-tree-row onClick={() => onClickRow(node)} className="flex-1 text-left font-mono text-xs text-text-muted hover:text-text-primary truncate">
      {node.label}
    </button>
    <button data-tree-eye data-dimmed={parentHiddenButSelfNot ? "true" : "false"} onClick={() => onToggleVisibility(node.id)} aria-label={isHidden ? "Show" : "Hide"} className={`text-text-ghost hover:text-accent ${parentHiddenButSelfNot ? "opacity-40" : ""}`}>
      {isHidden ? <EyeOff size={12}/> : <Eye size={12}/>}
    </button>
  </div>
  {hasChildren && isOpen && node.children!.map(c => <TreeRow key={c.id} node={c} ancestry={[...ancestry, node.id]} depth={depth+1} {...handlers}/>)}
```
