```tsx
import React from "react";
import { ChevronRight, ChevronDown, Eye, EyeOff } from "lucide-react";
import type { TreeNode } from "@/lib/buildRoomTree";
import { isHiddenInTree } from "@/lib/isHiddenInTree";

interface TreeRowProps {
  node: TreeNode;
  ancestry: string[]; // excluding self id
  depth: 0 | 1 | 2;
  expanded: Record<string, boolean>;
  activeRoomId: string | null;
  selectedIds: string[];
  hiddenIds: Set<string>;
  parentLabel?: string; // for cascade aria-label
  onToggleExpand: (id: string) => void;
  onClickRow: (node: TreeNode) => void;
  onToggleVisibility: (id: string) => void;
}

const INDENT: Record<0 | 1 | 2, string> = {
  0: "pl-2",
  1: "pl-4",
  2: "pl-6",
};

export function TreeRow(props: TreeRowProps) {
  const { node, ancestry, depth, expanded, activeRoomId, selectedIds, hiddenIds, parentLabel } = props;
  const isRoom = node.kind === "room";
  const isGroup = node.kind === "group";
  const isOpen = isRoom ? (expanded[node.id] ?? false) : true; // groups always open
  const selfHidden = hiddenIds.has(node.id);
  const cascadeHidden = isHiddenInTree(ancestry, hiddenIds);
  const parentOnlyHidden = cascadeHidden && !selfHidden;
  const selected = selectedIds.includes(node.id);
  const isActiveRoom = isRoom && node.id === activeRoomId;

  // Row classes — h-6 (24px), indent per depth, hover, selected state per UI-SPEC § Color
  const rowClass = [
    "group flex items-center h-6 pr-2 rounded-sm cursor-pointer",
    INDENT[depth],
    "hover:bg-obsidian-high",
    "focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-accent",
    selected ? "bg-obsidian-highest border-l-2 border-accent" : "",
    parentOnlyHidden ? "opacity-50" : "",
  ].filter(Boolean).join(" ");

  // Label classes per UI-SPEC § Typography + § Color
  const labelSize =
    depth === 0 ? "text-base font-medium" :
    depth === 1 ? "text-sm font-medium" :
    "text-sm";
  const labelColor =
    selected || isActiveRoom ? "text-accent-light" :
    parentOnlyHidden ? "text-text-ghost italic" :
    selfHidden ? "text-text-ghost" :
    "text-text-primary";
  const labelClass = `flex-1 text-left font-mono truncate overflow-hidden text-ellipsis whitespace-nowrap ${labelSize} ${labelColor}`;

  // Eye color tokens per UI-SPEC § Color visibility-state table
  const eyeColor =
    parentOnlyHidden ? "text-text-ghost opacity-50" :
    selfHidden ? "text-text-muted" :
    "text-text-dim hover:text-accent";

  // aria-label VERBATIM per UI-SPEC § Copywriting
  const eyeAria =
    parentOnlyHidden ? `${node.label} hidden because ${parentLabel ?? "parent"} is hidden` :
    selfHidden ? `Show ${node.label} in 3D view` :
    `Hide ${node.label} from 3D view`;

  const chevronAria = isOpen ? `Collapse ${node.label}` : `Expand ${node.label}`;

  return (
    <div data-tree-node={node.id} data-tree-kind={node.kind}>
      <div
        className={rowClass}
        {...(selected ? { "aria-current": "true" } : {})}
        onClick={() => {
          if (isGroup) return; // UI-SPEC § Interaction: group click NO-OP for selection
          props.onClickRow(node);
        }}
      >
        {isRoom ? (
          <button
            data-tree-chevron
            aria-label={chevronAria}
            aria-expanded={isOpen}
            className="text-text-ghost hover:text-accent focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-accent"
            onClick={(e) => {
              e.stopPropagation();
              props.onToggleExpand(node.id);
            }}
          >
            {isOpen ? <ChevronDown className="w-4 h-4" /> : <ChevronRight className="w-4 h-4" />}
          </button>
        ) : depth === 2 ? (
          // Leaf-row spacer to keep label column aligned with rooms (UI-SPEC § Per-Row Anatomy)
          <span className="w-4 h-4" aria-hidden="true" />
        ) : null}

        <span className="w-1" aria-hidden="true" /> {/* 4px gap-1 between chevron and label */}

        <button data-tree-row className={labelClass} onClick={(e) => {
          if (isGroup) { e.stopPropagation(); return; }
          // Bubble through to row onClick handler
        }}>
          {node.label}
        </button>

        <button
          data-tree-eye
          data-dimmed={parentOnlyHidden ? "true" : "false"}
          aria-label={eyeAria}
          className={`w-6 h-6 flex items-center justify-center focus-visible:outline-none focus-visible:ring-1 focus-visible:ring-accent ${eyeColor}`}
          onClick={(e) => {
            e.stopPropagation();
            props.onToggleVisibility(node.id);
          }}
        >
          {selfHidden || parentOnlyHidden ? <EyeOff className="w-3.5 h-3.5" /> : <Eye className="w-3.5 h-3.5" />}
        </button>
      </div>

      {/* Children: rooms render only when expanded; groups always render */}
      {isOpen && node.children && node.children.length > 0 && node.children.map((child) => (
        <TreeRow
          key={child.id}
          node={child}
          ancestry={[...ancestry, node.id]}
          depth={(depth + 1) as 0 | 1 | 2}
          expanded={expanded}
          activeRoomId={activeRoomId}
          selectedIds={selectedIds}
          hiddenIds={hiddenIds}
          parentLabel={node.label}
          onToggleExpand={props.onToggleExpand}
          onClickRow={props.onClickRow}
          onToggleVisibility={props.onToggleVisibility}
        />
      ))}

      {/* Empty-state row — UI-SPEC § Empty States VERBATIM. Group is "walls"/"products"/"custom" with empty children. */}
      {isOpen && isGroup && node.children && node.children.length === 0 && (
        <div className="flex items-center h-6 pl-6 pr-2 italic text-text-ghost font-mono text-sm">
          {node.groupKey === "walls" && "No walls yet"}
          {node.groupKey === "products" && "No products placed"}
          {node.groupKey === "custom" && "No custom elements placed"}
        </div>
      )}
    </div>
  );
}
```

```tsx
import React, { useEffect, useMemo, useState, useCallback } from "react";
import { CollapsibleSection } from "@/components/ui/CollapsibleSection";
import { TreeRow } from "./TreeRow";
import { buildRoomTree, type TreeNode } from "@/lib/buildRoomTree";
import { useCADStore } from "@/stores/cadStore";
import { useUIStore } from "@/stores/uiStore";
import type { Product } from "@/types/product";
import {
  focusOnRoom, focusOnWall, focusOnPlacedProduct,
  focusOnCeiling, focusOnPlacedCustomElement,
} from "./focusDispatch";

const STORAGE_PREFIX = "gsd:tree:room:";
const STORAGE_SUFFIX = ":expanded";

interface Props {
  productLibrary: Product[];
}

export function RoomsTreePanel({ productLibrary }: Props) {
  const rooms = useCADStore((s) => s.rooms);
  const customElements = useCADStore((s) => s.customElements);
  const activeRoomId = useCADStore((s) => s.activeRoomId);
  const selectedIds = useUIStore((s) => s.selectedIds);
  const hiddenIds = useUIStore((s) => s.hiddenIds);

  // Hydrate per-room expanded state from localStorage.
  // Default per UI-SPEC § Expand/collapse D-02: active room expanded, others collapsed.
  const [expanded, setExpanded] = useState<Record<string, boolean>>(() => {
    const out: Record<string, boolean> = {};
    Object.keys(rooms).forEach((id) => {
      const stored = typeof localStorage !== "undefined"
        ? localStorage.getItem(STORAGE_PREFIX + id + STORAGE_SUFFIX)
        : null;
      if (stored === "true") out[id] = true;
      else if (stored === "false") out[id] = false;
      else out[id] = id === activeRoomId; // D-02 default
    });
    return out;
  });

  // Reconcile when room set changes (room added/removed).
  useEffect(() => {
    setExpanded((prev) => {
      const next: Record<string, boolean> = {};
      Object.keys(rooms).forEach((id) => {
        if (id in prev) next[id] = prev[id];
        else {
          const stored = localStorage.getItem(STORAGE_PREFIX + id + STORAGE_SUFFIX);
          next[id] = stored === "true" || (stored === null && id === activeRoomId);
        }
      });
      return next;
    });
  }, [rooms, activeRoomId]);

  const tree = useMemo(
    () => buildRoomTree(rooms, customElements as Record<string, { id: string; name: string }>, productLibrary),
    [rooms, customElements, productLibrary],
  );

  const onToggleExpand = useCallback((id: string) => {
    setExpanded((prev) => {
      const nextValue = !prev[id];
      try {
        localStorage.setItem(STORAGE_PREFIX + id + STORAGE_SUFFIX, String(nextValue));
      } catch { /* ignore quota */ }
      return { ...prev, [id]: nextValue };
    });
  }, []);

  const onToggleVisibility = useCallback((id: string) => {
    useUIStore.getState().toggleHidden(id);
  }, []);

  const onClickRow = useCallback((node: TreeNode) => {
    const state = useCADStore.getState();
    const doc = state.rooms[node.roomId];
    if (!doc) return;
    if (node.kind === "room") {
      focusOnRoom(doc);
    } else if (node.kind === "wall") {
      const wall = doc.walls[node.id];
      if (wall) focusOnWall(wall);
    } else if (node.kind === "product") {
      const pp = doc.placedProducts[node.id];
      if (!pp) return;
      const product = productLibrary.find((p) => p.id === pp.productId);
      if (product) focusOnPlacedProduct(pp, product);
    } else if (node.kind === "ceiling") {
      const ceiling = (doc.ceilings ?? {})[node.id];
      if (ceiling) focusOnCeiling(doc, ceiling);
    } else if (node.kind === "custom") {
      const placed = (doc.placedCustomElements ?? {})[node.id];
      if (!placed) return;
      const catalog = (state.customElements as Record<string, { id: string; name: string; widthFt?: number; depthFt?: number; heightFt?: number }>)[
        (placed as { customElementId: string }).customElementId
      ];
      focusOnPlacedCustomElement(placed, catalog);
    }
  }, [productLibrary]);

  return (
    <CollapsibleSection label="Rooms" defaultOpen={true}>
      <div role="tree" aria-label="Rooms tree">
        {tree.map((roomNode) => (
          <TreeRow
            key={roomNode.id}
            node={roomNode}
            ancestry={[]}
            depth={0}
            expanded={expanded}
            activeRoomId={activeRoomId}
            selectedIds={selectedIds}
            hiddenIds={hiddenIds}
            onToggleExpand={onToggleExpand}
            onClickRow={onClickRow}
            onToggleVisibility={onToggleVisibility}
          />
        ))}
      </div>
    </CollapsibleSection>
  );
}
```

```typescript
declare global {

    __driveTreeVisibility?: (id: string) => void;
    __driveTreeSelect?: (id: string) => void;
    __getTreeState?: () => {

    };

export function installTreeDrivers(): void {
  if (typeof window === "undefined") return;
  if (import.meta.env.MODE !== "test") return;

  window.__driveTreeExpand = (roomId) => {
    const el = document.querySelector(`[data-tree-node="${roomId}"] [data-tree-chevron]`) as HTMLElement | null;
    el?.click();
  };
  window.__driveTreeVisibility = (id) => {
    const el = document.querySelector(`[data-tree-node="${id}"] [data-tree-eye]`) as HTMLElement | null;
    el?.click();
  };
  window.__driveTreeSelect = (id) => {
    const el = document.querySelector(`[data-tree-node="${id}"] [data-tree-row]`) as HTMLElement | null;
    el?.click();
  };
  window.__getTreeState = () => {
    const expanded: Record<string, boolean> = {};
    document.querySelectorAll('[data-tree-kind="room"]').forEach((el) => {
      const id = (el as HTMLElement).dataset.treeNode;
      const chevron = el.querySelector('[data-tree-chevron]');
      const expandedAttr = chevron?.getAttribute("aria-expanded");
      if (id) expanded[id] = expandedAttr === "true";
    });
    return {
      expanded,
      hiddenIds: Array.from(useUIStore.getState().hiddenIds),
      selectedIds: useUIStore.getState().selectedIds,
    };
  };
}
```
