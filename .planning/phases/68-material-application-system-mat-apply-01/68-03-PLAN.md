    ```typescript
    export async function migrateV5ToV6(snap: CADSnapshot): Promise<CADSnapshot> {
      if (snap.version >= 6) return snap;
      const customPaints = (snap as any).customPaints ?? [];

      for (const doc of Object.values(snap.rooms ?? {})) {
        // Floor
        if ((doc as any).floorMaterial && !(doc as any).floorMaterialId) {
          try {
            const id = await migrateFloorMaterialToMaterialId((doc as any).floorMaterial);
            if (id) (doc as any).floorMaterialId = id;
          } catch (err) {
            console.warn("[Phase68] floor migration failed ‚Äî entry preserved as legacy", err);
          }
        }
        // Walls
        for (const wall of Object.values((doc as any).walls ?? {}) as any[]) {
          const wp = wall.wallpaper;
          if (!wp) continue;
          for (const side of ["A", "B"] as const) {
            const w = wp[side];
            if (!w) continue;
            const field = side === "A" ? "materialIdA" : "materialIdB";
            if (wall[field]) continue;  // already migrated (idempotency)
            try {
              const id = await migrateWallpaperToMaterial(w, customPaints);
              if (id) wall[field] = id;
            } catch (err) {
              console.warn("[Phase68] wallpaper migration failed ‚Äî entry preserved as legacy", err);
            }
          }
        }
        // Ceilings
        for (const c of Object.values((doc as any).ceilings ?? {}) as any[]) {
          if (c.materialId) continue;  // idempotency
          try {
            const id = await migrateCeilingToMaterialId(c);
            if (id) c.materialId = id;
          } catch (err) {
            console.warn("[Phase68] ceiling migration failed ‚Äî entry preserved as legacy", err);
          }
        }
      }
      (snap as { version: number }).version = 6;
      return snap;
    }
    ```
