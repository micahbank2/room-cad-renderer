    ```typescript
    import * as fabric from "fabric";
    import type { Material } from "@/types/material";
    import { getUserTexture } from "@/lib/userTextureStore";  // VERIFY exact name

    const patternCache = new Map<string, fabric.Pattern>();
    const inflightLoads = new Map<string, Promise<fabric.Pattern | null>>();

    export function getMaterialPattern(
      material: Material,
      scaleFt: number,
      pixelsPerFoot: number,
      onReady?: (pattern: fabric.Pattern) => void,
    ): fabric.Pattern | null {
      if (!material.colorMapId) return null;
      const key = `${material.id}|${scaleFt}|${pixelsPerFoot}`;
      const hit = patternCache.get(key);
      if (hit) return hit;
      if (inflightLoads.has(key)) {
        if (onReady) inflightLoads.get(key)!.then((p) => p && onReady(p));
        return null;
      }
      const promise = (async (): Promise<fabric.Pattern | null> => {
        try {
          const ut = await getUserTexture(material.colorMapId!);
          if (!ut?.blob) return null;
          const blobUrl = URL.createObjectURL(ut.blob);
          const img = await new Promise<HTMLImageElement>((res, rej) => {
            const im = new Image();
            im.onload = () => res(im);
            im.onerror = rej;
            im.src = blobUrl;
          });
          const tileSizePx = Math.max(8, Math.round(scaleFt * pixelsPerFoot));
          const canvas = document.createElement("canvas");
          canvas.width = canvas.height = tileSizePx;
          const ctx = canvas.getContext("2d");
          if (!ctx) { URL.revokeObjectURL(blobUrl); return null; }
          ctx.drawImage(img, 0, 0, tileSizePx, tileSizePx);
          URL.revokeObjectURL(blobUrl);
          const pattern = new fabric.Pattern({ source: canvas, repeat: "repeat" });
          patternCache.set(key, pattern);
          inflightLoads.delete(key);
          if (onReady) onReady(pattern);
          return pattern;
        } catch (err) {
          console.warn("[Phase68] fabric pattern load failed", err);
          inflightLoads.delete(key);
          return null;
        }
      })();
      inflightLoads.set(key, promise);
      return null;
    }

    export function clearMaterialPatternCache(): void {
      patternCache.clear();
      inflightLoads.clear();
    }
    ```

       ```typescript
       export function renderFloor(
         fc: fabric.Canvas,
         room: any,  // RoomDoc
         scale: number,
         origin: { x: number; y: number },
         materials: ReadonlyArray<Material>,
       ): void {
         if (!room?.floorMaterialId) return;
         const mat = materials.find((m) => m.id === room.floorMaterialId);
         if (!mat) return;

         // Compute room footprint polygon from room.walls. For convex rooms: collect all wall.a points in order.
         const points = (Object.values(room.walls ?? {}) as any[]).map((w) => ({
           x: origin.x + w.a.x * scale,
           y: origin.y + w.a.y * scale,
         }));
         if (points.length < 3) return;

         let fill: string | fabric.Pattern = "#444";
         if (mat.colorHex) fill = mat.colorHex;
         else if (mat.colorMapId) {
           const tileFt = resolveSurfaceTileSize(room.floorScaleFt, mat);
           const pattern = getMaterialPattern(mat, tileFt, scale, () => fc.requestRenderAll());
           if (pattern) fill = pattern;
         }

         const poly = new fabric.Polygon(points, {
           fill: fill as any,
           selectable: false,
           evented: false,
           objectCaching: false,
         });
         fc.add(poly);
         fc.sendObjectToBack(poly);
       }
       ```
