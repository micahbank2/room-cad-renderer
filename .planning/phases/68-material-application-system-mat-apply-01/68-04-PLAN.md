    ```typescript
    import { useEffect, useMemo } from "react";
    import * as THREE from "three";
    import { resolveSurfaceMaterial, type ResolvedSurfaceMaterial } from "@/lib/surfaceMaterial";
    import { useMaterials } from "@/hooks/useMaterials";
    import { useUserTexture } from "@/three/userTextureCache";  // VERIFY exact import name

    export interface ResolvedSurfaceMaterialWithTextures
      extends Omit<ResolvedSurfaceMaterial, "colorMapId" | "roughnessMapId" | "reflectionMapId"> {
      colorMap?: THREE.Texture | null;
      roughnessMap?: THREE.Texture | null;
      reflectionMap?: THREE.Texture | null;
    }

    /**
     * Phase 68 â€” R3F wrapper around resolveSurfaceMaterial + userTextureCache.
     * Returns null when materialId unset or Material missing (caller falls back to legacy chain).
     */
    export function useResolvedMaterial(
      materialId: string | undefined,
      surfaceScaleFt: number | undefined,
      surfaceWidthFt: number,
      surfaceHeightFt: number,
    ): ResolvedSurfaceMaterialWithTextures | null {
      const { materials } = useMaterials();
      const resolved = useMemo(
        () => resolveSurfaceMaterial(materialId, surfaceScaleFt, materials),
        [materialId, surfaceScaleFt, materials],
      );

      const colorMap = useUserTexture(resolved?.colorMapId);
      const roughnessMap = useUserTexture(resolved?.roughnessMapId);
      const reflectionMap = useUserTexture(resolved?.reflectionMapId);

      // Apply tile-size repeat. Effect runs only when texture or tile size changes.
      useEffect(() => {
        if (!colorMap || !resolved) return;
        const repeatX = surfaceWidthFt / resolved.tileSizeFt;
        const repeatY = surfaceHeightFt / resolved.tileSizeFt;
        colorMap.wrapS = colorMap.wrapT = THREE.RepeatWrapping;
        colorMap.repeat.set(repeatX, repeatY);
        colorMap.needsUpdate = true;
        if (roughnessMap) {
          roughnessMap.wrapS = roughnessMap.wrapT = THREE.RepeatWrapping;
          roughnessMap.repeat.set(repeatX, repeatY);
          roughnessMap.needsUpdate = true;
        }
        if (reflectionMap) {
          reflectionMap.wrapS = reflectionMap.wrapT = THREE.RepeatWrapping;
          reflectionMap.repeat.set(repeatX, repeatY);
          reflectionMap.needsUpdate = true;
        }
      }, [colorMap, roughnessMap, reflectionMap, resolved, surfaceWidthFt, surfaceHeightFt]);

      if (!resolved) return null;
      return { ...resolved, colorMap, roughnessMap, reflectionMap };
    }
    ```

    ```typescript
    import type { FaceDirection } from "@/types/material";

    const FACE_ORDER: FaceDirection[] = ["east", "west", "top", "bottom", "north", "south"];
    // BoxGeometry face index order: [+X, -X, +Y, -Y, +Z, -Z]

    function FaceMaterial({
      faceName,
      placedMaterialId,
      fallbackColor,
      width, height,
    }: {
      faceName: FaceDirection;
      placedMaterialId: string | undefined;
      fallbackColor: string;
      width: number;
      height: number;
    }) {
      const resolved = useResolvedMaterial(placedMaterialId, undefined, width, height);
      if (resolved?.colorHex) {
        return <meshStandardMaterial attach="material" color={resolved.colorHex} roughness={resolved.roughness} metalness={resolved.metalness} />;
      }
      if (resolved?.colorMap) {
        return <meshStandardMaterial attach="material" map={resolved.colorMap} roughnessMap={resolved.roughnessMap ?? undefined} metalnessMap={resolved.reflectionMap ?? undefined} roughness={resolved.roughness} metalness={resolved.metalness} />;
      }
      return <meshStandardMaterial attach="material" color={fallbackColor} roughness={0.7} metalness={0} />;
    }

    export function CustomElementMesh({ element, placed, ... }: ...) {
      const faceMats = placed.faceMaterials ?? {};
      // Approximate per-face dimensions â€” use placed.width / placed.depth / placed.height
      const w = placed.widthFt ?? element.widthFt ?? 1;
      const d = placed.depthFt ?? element.depthFt ?? 1;
      const h = placed.heightFt ?? element.heightFt ?? 1;
      // Per-face surface size: top/bottom = wĂ—d, north/south = wĂ—h, east/west = dĂ—h

        top: [w, d], bottom: [w, d],
        north: [w, h], south: [w, h],

              faceName={face}
              placedMaterialId={faceMats[face]}
              fallbackColor={element.color}
              width={faceDims[face][0]}
              height={faceDims[face][1]}
            />
          ))}
        </mesh>
      );
    }
    ```
