    if (typeof window !== "undefined" && import.meta.env?.MODE === "test") {

        useCADStore.getState().applySurfaceMaterial(target, materialId);

        const { listMaterials } = await import("@/lib/materialStore");
        const { resolveSurfaceMaterial } = await import("@/lib/surfaceMaterial");
        const mats = await listMaterials();
        const state = useCADStore.getState();
        const doc: any = (state as any).rooms?.[(state as any).activeRoomId];
        if (!doc) return null;
        let mid: string | undefined;
        let scaleFt: number | undefined;
        switch (target.kind) {
          case "wallSide": {
            const w = doc.walls?.[target.wallId]; if (!w) return null;
            mid = target.side === "A" ? w.materialIdA : w.materialIdB;
            scaleFt = target.side === "A" ? w.scaleFtA : w.scaleFtB;
            break;

          case "floor": mid = doc.floorMaterialId; scaleFt = doc.floorScaleFt; break;
          case "ceiling": {
            const c = doc.ceilings?.[target.ceilingId]; if (!c) return null;
            mid = c.materialId; scaleFt = c.scaleFt; break;
          }
          case "customElementFace": {
            const p = doc.placedCustomElements?.[target.placedId]; if (!p) return null;
            mid = p.faceMaterials?.[target.face]; break;
          }
        }
        const r = resolveSurfaceMaterial(mid, scaleFt, mats);
        if (!r) return null;
        return { materialId: mid ?? null, hasColorHex: !!r.colorHex, hasColorMap: !!r.colorMapId };
      };
    }
