  renameWall: (roomId, wallId, name) => {
    pushHistory();
    set(produce((state) => {
      const doc = state.rooms[roomId];
      if (!doc) return;
      const wall = doc.walls[wallId];
      if (!wall) return;
      const trimmed = name.trim().slice(0, 40);
