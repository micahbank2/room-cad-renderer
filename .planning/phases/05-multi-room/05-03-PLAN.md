    **Create `src/components/AddRoomDialog.tsx`:**
    ```typescript
    import { useState } from "react";
    import { useCADStore } from "@/stores/cadStore";
    import { ROOM_TEMPLATES, type RoomTemplateId } from "@/data/roomTemplates";

    interface Props { open: boolean; onClose: () => void; }

    export default function AddRoomDialog({ open, onClose }: Props) {
      const [name, setName] = useState("");
      const [tpl, setTpl] = useState<RoomTemplateId>("BLANK");
      if (!open) return null;

      const handleCreate = () => {
        if (!name.trim()) return;
        useCADStore.getState().addRoom(name.trim(), tpl);
        setName(""); setTpl("BLANK");
        onClose();
      };

      const ids: RoomTemplateId[] = ["LIVING_ROOM", "BEDROOM", "KITCHEN", "BLANK"];

      return (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-obsidian-deepest/60 backdrop-blur-sm">
          <div className="bg-obsidian-mid border border-outline-variant/30 p-6 w-[480px] font-mono">
            <h2 className="text-accent-light text-sm mb-4">ADD_ROOM</h2>
            <label className="block text-text-dim text-xs mb-1">ROOM_NAME</label>
            <input
              autoFocus
              value={name}
              onChange={(e) => setName(e.target.value)}
              className="w-full bg-obsidian-deepest text-text-primary px-2 py-1 border border-outline-variant/30 mb-4 font-mono text-sm"
              placeholder="ROOM_NAME"
            />
            <label className="block text-text-dim text-xs mb-2">TEMPLATE</label>
            <div className="grid grid-cols-2 gap-2 mb-4">
              {ids.map((id) => (
                <button
                  key={id}
                  onClick={() => setTpl(id)}
                  className={`p-3 text-xs text-left border ${tpl === id ? "border-accent text-accent-light" : "border-outline-variant/30 text-text-muted hover:border-accent/50"}`}
                >
                  {ROOM_TEMPLATES[id].label}
                </button>
              ))}
            </div>
            <div className="flex justify-end gap-2">
              <button onClick={onClose} className="px-3 py-1 text-xs text-text-dim hover:text-text-muted">CANCEL</button>
              <button
                onClick={handleCreate}
                disabled={!name.trim()}
                className="px-3 py-1 text-xs bg-accent text-obsidian-deepest disabled:opacity-40"
              >
                CREATE
              </button>
            </div>
          </div>
        </div>
      );
    }
    ```
