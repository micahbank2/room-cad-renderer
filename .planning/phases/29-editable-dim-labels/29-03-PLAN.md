    ```ts
    const commit = () => {
      const parsed = parser ? parser(draft) : parseFloat(draft);

        setEditing(false);
        return;

      const v = parsed as number;
      // Min guard (existing)
      if (v < (min ?? 0)) {
        setEditing(false);
        return;
      }
      // No-op guard — suppresses round-trip float drift from formatFeet prefills (RESEARCH Pitfall #2)
      if (Math.abs(v - value) <= 1e-6) {
        setEditing(false);
        return;
      }
      onCommit(v);
      setEditing(false);
    };
    ```
