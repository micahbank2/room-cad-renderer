    // Phase 82 IA-05 — opening sub-inspector. Mounted by WallInspector when
    // uiStore.selectedOpeningId matches an opening on the selected wall.

    // Phase 79 invariants (verbatim — DO NOT MODIFY):
    // - D-07 single-undo: applyPreset → ONE update(wall.id, opening.id, ...)

    import { useCADStore } from "@/stores/cadStore";

      const setSelectedOpeningId = useUIStore((s) => s.setSelectedOpeningId);

      const isWindow = opening.type === "window";
      const defaultTab = isWindow ? "preset" : "type";
      const [activeTab, setActiveTab] = useState<string>(defaultTab);

      return (
        <div className="space-y-2" key={opening.id}>
          <button
            type="button"

            onClick={() => setSelectedOpeningId(null)}

          >

          </button>

          <div className="font-sans text-xs text-foreground">

                </>
              )}

            {isWindow && (
              <TabsContent value="preset">

              </TabsContent>

                <div className="space-y-1">

                    Type: {opening.type.charAt(0).toUpperCase() + opening.type.slice(1)}
                  </p>
                  {/* Future: hinge side / swing direction for doors. */}
                </div>
              </TabsContent>
            )}

            <TabsContent value="dimensions">
              <div className="space-y-1">
                <NumericRow label="Width" unit="ft" value={opening.width}
                  onPreview={(v) => updateNoHistory(wall.id, opening.id, { width: v })}
                  onCommit={(v) => update(wall.id, opening.id, { width: v })}
                />
                <NumericRow label="Height" unit="ft" value={opening.height}
                  placeholder={opening.type === "passthrough" ? "Wall height" : undefined}
                  onPreview={(v) => updateNoHistory(wall.id, opening.id, { height: v })}
                  onCommit={(v) => update(wall.id, opening.id, { height: v })}
                />
                <NumericRow label="Sill" unit="ft" value={opening.sillHeight}

    function WindowPresetBody({ wall, opening, update }: {
      wall: WallSegment; opening: Opening;
      update: (wallId: string, openingId: string, changes: Partial<Opening>) => void;
    }) {
      // VERBATIM lift from PropertiesPanel.OpeningSection.tsx L99-162.
      // D-07 / D-08 invariants preserved by NOT EDITING this body.
      const derivedId: WindowPresetId | "custom" = derivePreset({
        width: opening.width, height: opening.height, sillHeight: opening.sillHeight,
      });
      const derivedLabel = derivedId === "custom"
        ? "Custom"
        : (WINDOW_PRESETS.find((p) => p.id === derivedId)?.label ?? "Custom");

      function applyPreset(p: { width: number; height: number; sillHeight: number }) {
        update(wall.id, opening.id, { width: p.width, height: p.height, sillHeight: p.sillHeight });
      }

      return (
        <div className="flex flex-col gap-2">
          <div data-testid="opening-preset-label" className="text-sm font-sans text-muted-foreground">
            {`Preset: ${derivedLabel}`}
          </div>
          <div className="flex flex-wrap gap-1">
            {WINDOW_PRESETS.map((p) => (
              <button key={p.id} type="button"
                data-testid={`opening-preset-chip-${opening.id}-${p.id}`}
                onClick={() => applyPreset(p)}
                className={cn(
                  "px-2 py-0.5 text-xs font-sans rounded-smooth-md border transition-colors",
                  derivedId === p.id
                    ? "bg-accent/10 ring-1 ring-accent/40 border-transparent text-foreground"
                    : "border-border hover:bg-accent text-muted-foreground",
                )}>{p.label}</button>
            ))}
            <button type="button"
              data-testid={`opening-preset-chip-${opening.id}-custom`}
              onClick={() => { /* no-op */ }}
              className={cn(
                "px-2 py-0.5 text-xs font-sans rounded-smooth-md border transition-colors",
                derivedId === "custom"
                  ? "bg-accent/10 ring-1 ring-accent/40 border-transparent text-foreground"
                  : "border-border hover:bg-accent text-muted-foreground",
              )}>Custom</button>
          </div>
        </div>
      );
    }
