```typescript
if (hit) {
  const isMetaClick = (opt.e as MouseEvent).metaKey || (opt.e as MouseEvent).ctrlKey;

  if (isMetaClick) {
    // Multi-select: toggle the hit item in/out of selection
    const current = useUIStore.getState().selectedIds;
    if (current.includes(hit.id)) {
      // Deselect if already selected
      useUIStore.getState().select(current.filter((id) => id !== hit.id));
    } else {
      useUIStore.getState().addToSelection(hit.id);
    }
    // Do NOT start drag on meta-click (just adjust selection)
    state.dragging = false;
    state.dragId = null;
    state.dragType = null;
    return;
  }

  useUIStore.getState().select([hit.id]);
  // ... rest of existing code (ceiling check, drag setup)
}
```

The key behavior:
- `Cmd+click` (Mac) or `Ctrl+click` (Win) toggles an item in/out of the current selection
- Meta-click does NOT initiate drag â€” it only changes selection
- Normal click replaces the selection as before
- Clicking empty space still clears selection
